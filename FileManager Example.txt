#include "FileManager.h"
// -----------------------------------------------------------

	// In main.cpp

	int *n, p, *t;
	int** problem;

	FileManager::LoadFile("input.txt", &p, &n, &t, &problem);

	cout << p << endl;

	for (int i = 0; i < p; i++)
	{
		for (int j = 0; j < 2; j++)
		{
			if (j == 0)
				cout << n[i] << " " << t[i] << endl;
			else
			{
				for (int k = 0; k < n[i]; k++)
					cout << problem[i][k] << " ";

				cout << endl;
			}
		}
	}
